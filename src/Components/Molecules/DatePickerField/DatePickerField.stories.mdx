export const Template = (args) => {
  const [loading, setLoading] = useState(true);
  const [val, setVal] = useState(args.inputValue);
  useEffect(() => {
    let isUnmounting = false;
    const loadLocale = async () => {
      if (args.locale) {
        setLoading(true);
        importFnsLocaleFile(args.locale)
          .catch(alert)
          .finally(() => {
            if (!isUnmounting) {
              setLoading(false);
            }
          });
      } else {
        if (!isUnmounting) {
          setLoading(false);
        }
      }
    };
    loadLocale();
    return () => {
      isUnmounting = true;
    };
  }, [args.locale]);
  if (loading) {
    return (
      <>
        <Icon icon={['fal', 'spinner']} size='2x' />
        Loading locale
      </>
    );
  }
  return <DatePickerField {...args} inputValue={val} onChange={setVal} />;
};
